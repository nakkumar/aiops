#!/bin/bash
one=$(top -b -n2 -p 1 | fgrep "Cpu(s)" | tail -1 | awk  '{ print $2 }')
echo "val=$one"


index="linux" sourcetype="check-cpu" | where val > 50

mstat="em_metrics"

| rest /services/server/status count=0 splunk_server=* | dedup splunk_server | table splunk_server
